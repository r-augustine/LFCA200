| chmod     | -R                |
| chown     | -R                |
| fdisk     | -l, --list        |
| iostat    | -d; -c; -k        |
| ps        | -ef; -aux         |
| yum/dnf   | update; install; remove   |
| vmstat    | -a; -D, --disk-sum; -p, --partition; -w, --wide   |
